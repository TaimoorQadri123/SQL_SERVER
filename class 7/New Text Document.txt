use aptech
select * from std

create view selectdata as 
select std.name as studentName,std.department ,faculty.name as facultyName ,course.name as CourseName 
from
std
inner join 
faculty
on
std.f_id=faculty.id
inner join 
course
on
std.c_id =course.id

select * from selectdata
